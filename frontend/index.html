# Simple GET — mini chart PNG
curl "http://localhost:3000/v1/tradingview/mini-chart\
?symbol=BINANCE:BTCUSDT\
&interval=1D\
&width=600&height=300\
&theme=dark\
&key=YOUR_API_KEY" \
&theme=dark" \
  -o btcusdt-chart.png

# Advanced chart with indicators (v2 POST)
curl -X POST "http://localhost:3000/v2/tradingview/advanced-chart" \
  -d '{"symbol":"NASDAQ:MSFT","interval":"1W","range":"1Y"}' \
  -o tesla-4h.png
// JavaScript (fetch)
const response = await fetch(
  'http://localhost:3000/v2/tradingview/advanced-chart',
  {
    method: 'POST',
    headers: {
      'x-api-key': 'YOUR_API_KEY',
      'Content-Type': 'application/json'
    },
    body: JSON.stringify({
      symbol: 'BINANCE:ETHUSDT',
      theme: 'dark',
      interval: '1h',
      width: 1200,
      height: 700,
      studies: [
        { name: 'Bollinger Bands' },
        { name: 'Volume', forceOverlay: true }
      ]
    })
  }
);
const blob = await response.blob();
const url = URL.createObjectURL(blob);

import requests

# Advanced chart via POST
response = requests.post(
  "http://localhost:3000/v2/tradingview/advanced-chart",
  headers={
    "x-api-key": "YOUR_API_KEY",
    "Content-Type": "application/json"
  },
  json={
    "symbol": "NASDAQ:MSFT",
    "theme": "dark",
    "interval": "1W",
    "style": "candle",
    "studies": [
      {"name": "Moving Average"},
      {"name": "Relative Strength Index"}
    ]
  }
)

with open("msft-chart.png", "wb") as f:
  f.write(response.content)
print("Chart saved!")
❓ FAQ

Frequently asked
questions

What chart types are supported? ⌄

We support all TradingView chart types: Candlestick, Bar, Line, Area, Heikin Ashi, Hollow Candle, Baseline, Hi-Lo, and Column. Set the style parameter to any of these values.

Which technical indicators are available? ⌄

Over 90 indicators are supported, including RSI, MACD, Bollinger Bands, EMA, SMA, Volume Profile, Ichimoku Cloud, Stochastic, ATR, ADX, CCI, and many more. See the full list in the documentation.

How do I authenticate my requests? ⌄

You can authenticate using three methods: include your key as an x-api-key header, as Authorization: Bearer YOUR_KEY, or as a ?key=YOUR_KEY query parameter.

What happens when I exceed my daily limit? ⌄

You will receive a 429 Too Many Requests response with a JSON error explaining your limit and current usage. Limits reset each day at midnight UTC.

Can I get chart images in JPEG format? ⌄

Yes! The v2 API (POST) supports both PNG and JPEG via the format parameter. PNG is the default for best quality. JPEG is smaller in file size and can be useful for messaging platforms.

Are Crypto, Forex, and Stock symbols supported? ⌄

Yes, all TradingView symbols are supported using the EXCHANGE:SYMBOL format (e.g., BINANCE:BTCUSDT, NASDAQ:AAPL, OANDA:EURUSD, NSE:RELIANCE). Use the v3 exchange endpoints to list available symbols.
